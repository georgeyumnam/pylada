[docs]def rock_salt():
  """ rock_salt lattice """
  from pylada.crystal import Structure
  return Structure( 1, 0, 0,\
                    0, 1, 0,\
                    0, 0, 1,\
                    scale=1, name='Rock-Salt' )\
           .add_atom(0, 0, 0, 'A')\
           .add_atom(0.5, 0.5, 0.5, 'B')

  """ wurtzite lattice """
